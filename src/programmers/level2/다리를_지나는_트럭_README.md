   public int solution(int bridge_length, int weight, int[] truck_weights) {
      Queue<Integer> queue = new ArrayDeque<>();

      for (int i = 0; i < bridge_length - 1; i++) {

      int result = 1;

            queue.offer(truck_weights[idx]);
            curWeight += truck_weights[idx];

         }

         queue.offer(0);
      }

      return result;
   }
